'use strict';
 
var _recursion1 = require('../recursion1');
 
describe('factorial - ', function () {
    it('Should throw an error when input is not a number', function () {
        expect(function () {
            return (0, _recursion1.factorial)('invalid');
        }).to['throw'](/n must be a number/);
    });
 
    it('Should throw an error when input is less then 0', function () {
        expect(function () {
            return (0, _recursion1.factorial)(-1);
        }).to['throw'](/n must be greater then 0/);
    });
 
    it('Should return 1 with no input', function () {
        var result = (0, _recursion1.factorial)();
        expect(result).to.equal(1);
    });
 
    it('Should return 1 when input is 0', function () {
        var result = (0, _recursion1.factorial)(0);
        expect(result).to.equal(1);
    });
 
    it('Should return 1 when input is 1', function () {
        var result = (0, _recursion1.factorial)(1);
        expect(result).to.equal(1);
    });
 
    it('Should return factorial when input is valid', function () {
        var result = (0, _recursion1.factorial)(5);
        expect(result).to.equal(120);
    });
});
